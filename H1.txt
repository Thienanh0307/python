   int i;

int lonThu2(int a[], int n){
    int lonNhat, lonNhi;
    if (a[0] > a[1]){
        lonNhat = a[0];
        lonNhi = a[1];
    }else{
        lonNhat = a[1];
        lonNhi = a[0];
    }
    for(int i = 2; i < n; i++){
        if(a[i] >= lonNhat){
            lonNhi = lonNhat;
            lonNhat = a[i];
        }else if (a[i] > lonNhi){
            lonNhi = a[i];
        }
    }
    return lonNhi;
}
